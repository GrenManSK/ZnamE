39 def get_size1(bytes: int | float, suffix: str = "B"):
40 """
41 It takes a number of bytes and returns a string with the number of bytes and the appropriate
42 unit
43
44 :param bytes: The number of bytes to convert
45 :param suffix: The suffix to be appended to the size, defaults to B (optional)
46 """
47 """
48 Scale bytes to its proper format
49 e.g:
50 1253656 => '1.20MB'
51 1253656678 => '1.17GB'
52 """
53 factor: int = 1024
54 for unit in ["", "K", "M", "G", "T", "P"]:
55 if bytes < factor:
56 return f"{bytes:.2f}{unit}{suffix}"
57 bytes /= factor
